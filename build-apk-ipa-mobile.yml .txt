name: Build All Platforms (Kivy)

on:
  push:
    branches: [ master, main ]
  workflow_dispatch:

jobs:
  # 1. بناء الأندرويد (APK + AAB)
  build-android:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build with Buildozer
        uses: ArtemSBulgakov/buildozer-action@v1
        id: buildozer
        with:
          buildozer_version: stable
      - name: Upload APK/AAB
        uses: actions/upload-artifact@v4
        with:
          name: android-builds
          path: bin/* # سيقوم برفع أي ملف APK أو AAB يتم بناؤه

  # 2. بناء ويندوز (Windows Exe)
  build-windows:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v4
      - name: Install PyInstaller
        run: |
          pip install pyinstaller kivy
      - name: Build Exe
        run: pyinstaller --onefile --windowed main.py
      - name: Upload Windows
        uses: actions/upload-artifact@v4
        with:
          name: windows-app
          path: dist/*.exe

  # 3. بناء لينكس (Linux AppImage/Binary)
  build-linux:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build Linux
        run: |
          pip install pyinstaller
          pyinstaller --onefile --windowed main.py
      - name: Upload Linux
        uses: actions/upload-artifact@v4
        with:
          name: linux-app
          path: dist/
